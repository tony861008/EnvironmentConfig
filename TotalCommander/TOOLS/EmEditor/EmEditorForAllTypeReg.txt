REGEDIT4

// 功能:在鼠标右键弹出菜单中为所有文件类型添加EmEditor编辑
//      选项,并设置为默认操作
// 提示:HKEY_CLASSES_ROOT主键下包含所有文件关联,
//      为某类文件添加命令均可参照此例

[HKEY_CLASSES_ROOT\*\shell]
@="EmEditor for all type"

[HKEY_CLASSES_ROOT\*\shell\EmEditor for all type]
@="EmEditor"

[HKEY_CLASSES_ROOT\*\shell\EmEditor for all type\command]
@="%EmEditorForAllTypeReg%\\emeditor.exe \"%1\""
